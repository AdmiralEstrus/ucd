
GUIEditor = {
    tab = {},
    tabpanel = {},
    button = {},
    window = {},
    gridlist = {}
}
addEventHandler("onClientResourceStart", resourceRoot,
    function()
        GUIEditor.window[1] = guiCreateWindow(653, 312, 600, 443, "UCD | Groups - Blacklist", false)
        guiWindowSetSizable(GUIEditor.window[1], false)

        GUIEditor.tabpanel[1] = guiCreateTabPanel(9, 24, 581, 372, false, GUIEditor.window[1])

        GUIEditor.tab[1] = guiCreateTab("Account", GUIEditor.tabpanel[1])

        GUIEditor.gridlist[1] = guiCreateGridList(9, 6, 562, 280, false, GUIEditor.tab[1])
        guiGridListAddColumn(GUIEditor.gridlist[1], "Account", 0.2)
        guiGridListAddColumn(GUIEditor.gridlist[1], "Reason", 0.2)
        guiGridListAddColumn(GUIEditor.gridlist[1], "Date", 0.2)
        guiGridListAddColumn(GUIEditor.gridlist[1], "Blacklisted by", 0.2)
        GUIEditor.button[1] = guiCreateButton(9, 296, 130, 38, "Add New", false, GUIEditor.tab[1])
        guiSetProperty(GUIEditor.button[1], "NormalTextColour", "FFAAAAAA")
        GUIEditor.button[2] = guiCreateButton(156, 296, 130, 38, "Delete Selected", false, GUIEditor.tab[1])
        guiSetProperty(GUIEditor.button[2], "NormalTextColour", "FFAAAAAA")

        GUIEditor.tab[2] = guiCreateTab("Serial", GUIEditor.tabpanel[1])

        GUIEditor.gridlist[2] = guiCreateGridList(9, 6, 562, 280, false, GUIEditor.tab[2])
        guiGridListAddColumn(GUIEditor.gridlist[2], "Serial", 0.2)
        guiGridListAddColumn(GUIEditor.gridlist[2], "Reason", 0.2)
        guiGridListAddColumn(GUIEditor.gridlist[2], "Date", 0.2)
        guiGridListAddColumn(GUIEditor.gridlist[2], "Blacklisted by", 0.2)
        GUIEditor.button[3] = guiCreateButton(9, 296, 130, 38, "Add New", false, GUIEditor.tab[2])
        guiSetProperty(GUIEditor.button[3], "NormalTextColour", "FFAAAAAA")
        GUIEditor.button[4] = guiCreateButton(156, 296, 130, 38, "Delete Selected", false, GUIEditor.tab[2])
        guiSetProperty(GUIEditor.button[4], "NormalTextColour", "FFAAAAAA")


        GUIEditor.button[5] = guiCreateButton(9, 402, 581, 31, "Close", false, GUIEditor.window[1])
        guiSetProperty(GUIEditor.button[5], "NormalTextColour", "FFAAAAAA")    
    end
)
