
GUIEditor = {
    gridlist = {},
    window = {},
    button = {}
}
addEventHandler("onClientResourceStart", resourceRoot,
    function()
        GUIEditor.window[1] = guiCreateWindow(674, 406, 519, 281, "UCD | Groups - Ranks", false)
        guiWindowSetSizable(GUIEditor.window[1], false)

        GUIEditor.gridlist[1] = guiCreateGridList(10, 25, 186, 201, false, GUIEditor.window[1])
        guiGridListAddColumn(GUIEditor.gridlist[1], "Rank Name", 0.9)
        for i = 1, 6 do
            guiGridListAddRow(GUIEditor.gridlist[1])
        end
        guiGridListSetItemText(GUIEditor.gridlist[1], 0, 1, "Trial", false, false)
        guiGridListSetItemText(GUIEditor.gridlist[1], 1, 1, "Regular", false, false)
        guiGridListSetItemText(GUIEditor.gridlist[1], 2, 1, "Trusted", false, false)
        guiGridListSetItemText(GUIEditor.gridlist[1], 3, 1, "Deputy", false, false)
        guiGridListSetItemText(GUIEditor.gridlist[1], 4, 1, "Leader", false, false)
        guiGridListSetItemText(GUIEditor.gridlist[1], 5, 1, "Founder", false, false)
        GUIEditor.button[1] = guiCreateButton(10, 236, 117, 35, "Add Rank", false, GUIEditor.window[1])
        guiSetProperty(GUIEditor.button[1], "NormalTextColour", "FFAAAAAA")
        GUIEditor.button[2] = guiCreateButton(137, 237, 117, 34, "Edit Rank", false, GUIEditor.window[1])
        guiSetProperty(GUIEditor.button[2], "NormalTextColour", "FFAAAAAA")
        GUIEditor.button[3] = guiCreateButton(264, 237, 117, 34, "Remove Rank", false, GUIEditor.window[1])
        guiSetProperty(GUIEditor.button[3], "NormalTextColour", "FFAAAAAA")
        GUIEditor.button[4] = guiCreateButton(391, 237, 117, 34, "Close", false, GUIEditor.window[1])
        guiSetProperty(GUIEditor.button[4], "NormalTextColour", "FFAAAAAA")    
    end
)
